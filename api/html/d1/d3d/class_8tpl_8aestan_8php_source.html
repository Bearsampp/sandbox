241 private static function getActionService($service, $action, $item = false)
242 {
243 global $bearsamppLang;
244 $result = 'Action: ' . $action;
245
246 if ($service != null) {
247 $result = 'Action: service; ' .
248 'Service: ' . $service . '; ' .
249 'ServiceAction: ' . $action;
250 }
251
252 if ($item) {
253 $result = 'Type: item; ' . $result;
254 if ($action == self::SERVICE_START) {
255 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_START_SERVICE) . '"' .
256 '; Glyph: ' . self::GLYPH_START;
257 } elseif ($action == self::SERVICE_STOP) {
258 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_STOP_SERVICE) . '"' .
259 '; Glyph: ' . self::GLYPH_STOP;
260 } elseif ($action == self::SERVICE_RESTART) {
261 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_RESTART_SERVICE) . '"' .
262 '; Glyph: ' . self::GLYPH_RELOAD;
263 }
264 } elseif ($action != self::SERVICES_CLOSE) {
265 $result .= '; Flags: ignoreerrors waituntilterminated';
266 }
267
268 return $result;
269 }
