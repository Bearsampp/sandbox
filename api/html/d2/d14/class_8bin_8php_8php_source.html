96 Util::logReloadClass($this);

102 $this->enable = $this->enable && $bearsamppConfig->getRaw(self::ROOT_CFG_ENABLE);
103 $this->apacheConf = $bearsamppBins->getApache()->getCurrentPath() . '/' . $this->apacheConf; //FIXME: Useful ?

106 if ($this->bearsamppConfRaw !== false) {

112
113 if (!$this->enable) {
114 Util::logInfo($this->name . ' is not enabled!');
115 return;
116 }
117 if (!is_dir($this->currentPath)) {
118 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_FILE_NOT_FOUND), $this->name . ' ' . $this->version, $this->currentPath));
119 return;
120 }
121 if (!is_dir($this->symlinkPath)) {
122 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_FILE_NOT_FOUND), $this->name . ' ' . $this->version, $this->symlinkPath));
123 return;
124 }
125 if (!is_file($this->bearsamppConf)) {
126 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_CONF_NOT_FOUND), $this->name . ' ' . $this->version, $this->bearsamppConf));
127 return;
128 }
129 if (!is_file($this->cliExe)) {
130 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_EXE_NOT_FOUND), $this->name . ' ' . $this->version, $this->cliExe));
131 }
132 if (!is_file($this->cliSilentExe)) {
133 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_EXE_NOT_FOUND), $this->name . ' ' . $this->version, $this->cliSilentExe));
134 }
135 if (!is_file($this->conf)) {
136 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_CONF_NOT_FOUND), $this->name . ' ' . $this->version, $this->conf));
137 }
138 if (!is_file($this->pearExe)) {
139 Util::logError(sprintf($bearsamppLang->getValue(Lang::ERROR_EXE_NOT_FOUND), $this->name . ' ' . $this->version, $this->pearExe));
140 }

148 protected function updateConfig($version = null, $sub = 0, $showWindow = false) {
149 global $bearsamppLang, $bearsamppBins, $bearsamppApps, $bearsamppWinbinder;
150
151 if (!$this->enable) {
152 return true;
153 }
154
155 $version = $version == null ? $this->version : $version;
156 Util::logDebug(($sub > 0 ? str_repeat(' ', 2 * $sub) : '') . 'Update ' . $this->name . ' ' . $version . ' config');
157
158 $boxTitle = sprintf($bearsamppLang->getValue(Lang::SWITCH_VERSION_TITLE), $this->getName(), $version);
159
160 //$phpPath = str_replace('php' . $this->getVersion(), 'php' . $version, $this->getCurrentPath());
161 $conf = str_replace('php' . $this->getVersion(), 'php' . $version, $this->getConf());
162 $bearsamppConf = str_replace('php' . $this->getVersion(), 'php' . $version, $this->bearsamppConf);
163
164 $tsDll = $this->getTsDll($version);
165 //$apacheShortVersion = substr(str_replace('.', '', $bearsamppBins->getApache()->getVersion()), 0, 2);
166 //$apachePhpModuleName = $tsDll !== false ? substr($tsDll, 0, 4) . '_module' : null;
167 $apachePhpModulePath = $this->getApacheModule($bearsamppBins->getApache()->getVersion(), $version);
168
169 Util::logDebug(($sub > 0 ? str_repeat(' ', 2 * $sub) : '') . 'PHP TsDll found: ' . $tsDll);
170 Util::logDebug(($sub > 0 ? str_repeat(' ', 2 * $sub) : '') . 'PHP Apache module found: ' . $apachePhpModulePath);
171
172 if (!file_exists($conf) || !file_exists($bearsamppConf)) {
173 Util::logError('bearsampp config files not found for ' . $this->getName() . ' ' . $version);
174 if ($showWindow) {
175 $bearsamppWinbinder->messageBoxError(
176 sprintf($bearsamppLang->getValue(Lang::BEARSAMPP_CONF_NOT_FOUND_ERROR), $this->getName() . ' ' . $version),
177 $boxTitle
178 );
179 }
180 return false;
181 }
182
183 $bearsamppConfRaw = parse_ini_file($bearsamppConf);
184 if ($bearsamppConfRaw === false || !isset($bearsamppConfRaw[self::ROOT_CFG_VERSION]) || $bearsamppConfRaw[self::ROOT_CFG_VERSION] != $version) {
185 Util::logError('bearsampp config file malformed for ' . $this->getName() . ' ' . $version);
186 if ($showWindow) {
187 $bearsamppWinbinder->messageBoxError(
188 sprintf($bearsamppLang->getValue(Lang::BEARSAMPP_CONF_MALFORMED_ERROR), $this->getName() . ' ' . $version),
189 $boxTitle
190 );
191 }
192 return false;
193 }
194
195 if ($tsDll === false || $apachePhpModulePath === false) {
196 Util::logDebug($this->getName() . ' ' . $version . ' does not seem to be compatible with Apache ' . $bearsamppBins->getApache()->getVersion());
197 if ($showWindow) {
198 $bearsamppWinbinder->messageBoxError(
199 sprintf($bearsamppLang->getValue(Lang::PHP_INCPT), $version, $bearsamppBins->getApache()->getVersion()),
200 $boxTitle
201 );
202 }
203 return false;
204 }
205
206 // bearsampp.conf
207 $this->setVersion($version);
208
209 // conf
210 Util::replaceInFile($this->getConf(), array(
211 '/^mysql.default_port\s=\s(\d+)/' => 'mysql.default_port = ' . $bearsamppBins->getMysql()->getPort(),
212 '/^mysqli.default_port\s=\s(\d+)/' => 'mysqli.default_port = ' . $bearsamppBins->getMysql()->getPort()
213 ));
214
215 // apache
216 $bearsamppBins->getApache()->update($sub + 1);
217
218 // phpmyadmin
219 $bearsamppApps->getPhpmyadmin()->update($sub + 1);
220
221 return true;
222 }

413 public function isSettingActive($name) {
414 $settingsValues = $this->getSettingsValues();
415
416 $confContent = file($this->getConf());
417 foreach ($confContent as $row) {
418 $settingMatch = array();
419 if (preg_match('/^' . $name . '\s*=\s*(.+)/i', $row, $settingMatch)) {
420 return isset($settingMatch[1]) && isset($settingsValues[$name]) && $settingsValues[$name][0] == trim($settingMatch[1]);
421 }
422 }
423
424 return false;
425 }
