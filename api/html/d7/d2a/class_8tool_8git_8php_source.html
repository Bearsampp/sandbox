115
116 if ($cache) {
117 if (file_exists($this->reposCacheFile)) {
118 $repos = file($this->reposCacheFile);
119 foreach ($repos as $repo) {

123 } else {

127 if (!empty($foundRepos)) {
128 foreach ($foundRepos as $foundRepo) {
129 array_push($result, $foundRepo);
130 }
131 }
132 }
133 }
134 $strResult = implode(PHP_EOL, $result);
135 file_put_contents($this->reposCacheFile, $strResult);
136 }
137
