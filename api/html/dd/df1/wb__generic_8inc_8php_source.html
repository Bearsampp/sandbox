39/* Returns an array with all files of subdirectory $path. If $subdirs is TRUE,
40 includes subdirectories recursively. $mask is a PCRE regular expression.
41*/
42
43function get_folder_files($path, $subdirs=false, $fullname=true, $mask="", $forcelowercase=TRUE)
44{
45 // Correct path name, if needed
46
47 $path = str_replace('/', '\\', $path);
48 if(substr($path, -1) != '\\')
49 $path .= "\\";
50 if(!$path || !@is_dir($path))
51 return array();
52
53 // Browse the subdiretory list recursively
54
55 $dir = array();
56 if($handle = opendir($path)) {
57 while(($file = readdir($handle)) !== false) {
58 if(!is_dir($path.$file)) { // No directories / subdirectories
59 if($forcelowercase)
60 $file = strtolower($file);
61 if(!$mask) {
62 $dir[] = $fullname ? $path.$file : $file;
63 } else if($mask && preg_match($mask, $file)) {
64 $dir[] = $fullname ? $path.$file : $file;
65 }
66 } else if($subdirs && $file[0] != ".") { // Exclude "." and ".."
67 $dir = array_merge($dir, get_folder_files($path.$file, $subdirs, $fullname, $mask));
68 }
69 }
70 }
71 closedir($handle);
72 return $dir;
73}

77/* Transforms the array $data in a text that can be saved as an INI file.
78 Escapes double-quotes as (\") */
79
80function generate_ini($data, $comments="")
81{
82 if(!is_array($data)) {
83 trigger_error(__FUNCTION__ . ": Cannot save INI file.");
84 return null;
85 }
86 $text = $comments;
87 foreach($data as $name=>$section) {
88 $text .= "\r\n[$name]\r\n";
89
90 foreach($section as $key=>$value) {
91 $value = trim($value);
92 if((string)((int)$value) == (string)$value) // Integer: does nothing
93 ;
94 elseif((string)((float)$value) == (string)$value) // Floating point: does nothing
95 ;
96 elseif($value === "") // Empty string
97 $value = '""';
98 elseif(strstr($value, '"')) // Escape double-quotes
99 $value = '"' . str_replace('"', '\"', $value) . '"';
100 else
101 $value = '"' . $value . '"';
102
103 $text .= "$key = " . $value . "\r\n";
104 }
105 }
106 return $text;
107}

116function parse_ini($initext, $changecase=TRUE, $convertwords=TRUE)
117{
118 $ini = preg_split("/\r\n|\n/", $initext);
119 $secpattern = "/^\[(.[^\]]*)\]/i";
120// $entrypattern = "/^([a-z_0-9]*)\s*=\s*\"?([^\"]*)?\"?" . '$' . "/i";
121// $strpattern = "/^\"?(.[^\"]*)\"?" . '$' . "/i";
122 $entrypattern = "/^([a-z_0-9]*)\s*=\s*\"?([^\"]*)?\"?\$/i";
123 $strpattern = "/^\"?(.[^\"]*)\"?\$/i";
124
125 $section = array();
126 $sec = "";
127
128 // Predefined words
129
130 static $words = array("yes", "on", "true", "no", "off", "false", "null");
131 static $values = array( 1, 1, 1, 0, 0, 0, null);
132
133 // Lines loop
134
135 for($i = 0; $i < count($ini); $i++) {
136
137 $line = trim($ini[$i]);
138
139 // Replaces escaped double-quotes (\") with special signal /%quote%/
140
141 if(strstr($line, '\"'))
142 $line = str_replace('\"', '/%quote%/', $line);
143
144 // Skips blank lines and comments
145
146 if($line == "" || preg_match("/^;/i", $line))
147 continue;
148
149 if(preg_match($secpattern, $line, $matches)) {
150
151 // It's a section
152
153 $sec = $matches[1];
154
155 if($changecase)
156 $sec = ucfirst(strtolower($sec));
157
158 $section[$sec] = array();
159
160 } elseif(preg_match($entrypattern, $line, $matches)) {
161
162 // It's an entry
163
164 $entry = $matches[1];
165
166 if($changecase)
167 $entry = strtolower($entry);
168
169 $value = preg_replace($entrypattern, "\\2", $line);
170
171 // Restores double-quotes (")
172
173 $value = str_replace('/%quote%/', '"', $value);
174
175 // Convert some special words to their respective values
176
177 if($convertwords) {
178 $index = array_search(strtolower($value), $words);
179 if($index !== false)
180 $value = $values[$index];
181 }
182
183 $section[$sec][$entry] = $value;
184
185 } else {
186
187 // It's a normal string
188
189 $section[$sec][] = preg_replace($strpattern, "\\1", $line);
190
191 }
192 }
193 return $section;
194}
